# Cloudinary Configuration
# Copy this content to a .env file in your project root and fill in your actual values

REACT_APP_CLOUDINARY_CLOUD_NAME=your_cloud_name_here
REACT_APP_CLOUDINARY_UPLOAD_PRESET=your_upload_preset_name_here
REACT_APP_CLOUDINARY_API_KEY=your_api_key_here
REACT_APP_CLOUDINARY_API_SECRET=your_api_secret_here

# Note: Only REACT_APP_CLOUDINARY_CLOUD_NAME and REACT_APP_CLOUDINARY_UPLOAD_PRESET 
# are required for client-side uploads. The API key and secret are for server-side operations.
